create table scene(
id_scene int not null primary key,
title_scene varchar(30),
dc_scene longtext
);

create table item(
id_item int not null primary key,
name_item varchar(20),
dc_item longtext,
item_result longtext,
correct_cmd varchar(30),
id_scene int,
id_next_scene int,
foreign key (id_scene) references scene (id_scene)
);

create table save(
id_save int primary key,
id_scene int,
foreign key (id_scene) references scene (id_scene)
);

create table inventory(
id_save int,
id_item_i int,
foreign key (id_save) references save (id_save),
foreign key (id_item_i) references item (id_item)
);
